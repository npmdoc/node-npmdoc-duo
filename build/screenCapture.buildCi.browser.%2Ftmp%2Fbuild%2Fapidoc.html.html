    function Duo(root) {
  if (!(this instanceof Duo)) return new Duo(root);
  if (!root) throw error('duo requires a root directory');
  Emitter.call(this);
  this.root(root);
  this.manifest('component.json');
  this.installTo('components');
  this.buildTo('build');
  this.copy(false);
  this.development(false);
  this.sourceMap(false);
  this.concurrency(50);
  this.update(false);
  this.cache(true);

  this.assets = [];
  this.mapping = {};
  this.includes = {};
  this.packages = {};
  this.json = readJson(this.path('component.json'));

  // plugins
  this.plugins = new Ware();
  this.plugins.run = thunk(this.plugins.run);

  // alternate plugins (used for the full build, not individual files)
  this.altPlugins = new Ware();
  this.altPlugins.run = thunk(this.altPlugins.run);
}

    function File(attrs, duo) {
  if (!(this instanceof File)) return new File(attrs, duo);
  this.attrs = clone(attrs || {});
  this.duo = duo;

  var type = attrs.type ? '.' + attrs.type : '';
  var path = attrs.path || 'source' + type;

  // initial attrs
  this.attrs.path = resolve(attrs.root, path);
  this.attrs.type = attrs.type || extension(this.path);
  this.attrs.id = relative(duo.root(), this.path);

  // local vs remote
  this.attrs.local = !!(attrs.entry || attrs.local);
}

function duo.util.findroot (root)

  • description and source-code
    findroot = function (root) {
  var cwd = process.cwd();
  if (root) return resolve(cwd, root);
  var sep = path.sep;
  var parts = cwd.split(sep);
  var dir = cwd;

  while (!exists(path.join(dir, 'component.json')) && parts.length > 1) {
    parts.pop();
    dir = parts.join(sep);
  }

  return parts.length <= 1
    ? cwd
    : dir;
}

function duo.util.plugins (root, plugins)

  • description and source-code
    plugins = function (root, plugins) {
  return plugins.map(function (plugin) {
    var local = resolve(root, plugin);
    var npm = resolve(root, 'node_modules', plugin);
    var cwd = resolve(process.cwd(), 'node_modules', plugin);
    var mod;

    if (exists(local)) mod = require(local);
    else if (exists(local + '.js')) mod = require(local);
    else if (exists(npm)) mod = require(npm);
    else mod = require(cwd);

    return Array.isArray(mod) ? mod : mod();
  }, []);
}
